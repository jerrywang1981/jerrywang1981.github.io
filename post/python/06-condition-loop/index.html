#!/usr/bin/env python3
# -*- coding: utf-8 -*-

if __name__ == '__main__':
    # while condition
    a = 5
    while a > 0:
        # print(a)
        a = a -1
    # for in
    for i in range(10):
        print(i)

    # break the loop
    
    print('------------------')
    for i in range(10):
        print(i)
    else: # 
        print('done')
        
    # continue the loop
    print('------------------')
    for i in range(10):
        if i == 5:
            continue 
        print(i)
    count = 0
    while count < 5:
       print (count, " 小于 5")
       count = count + 1
    else:
       print (count, " 大于或等于 5")
