原型模式

package prototype_pattern

type Person struct {
	Name string
	Age  int
}

type Cloner interface {
	Clone() interface{}
}

func (s *Person) Clone() interface{} {
	var n Person
	n.Name = s.Name
	n.Age = s.Age
	return n
}
